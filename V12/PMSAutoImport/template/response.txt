        function adjustRemainingTimeoutSeconds() {
            if (!timeoutWindowIsDisplayed) return;
            var secondsLeft = document.getElementById('SessionExpiringInSeconds');
            var timeLeft = document.getElementById('SessionExpiringIn');

            var seconds = parseInt(secondsLeft.innerHTML);
            seconds--;
            secondsLeft.innerHTML = seconds;
            var s = seconds % 60;
            timeLeft.innerHTML = Math.floor(seconds / 60) + ':' + (s > 9 ? s : '0' + s);

            if (seconds > 0) {
                setTimeout('adjustRemainingTimeoutSeconds()', 1000);
            } else {
                $('#divSessionExpiring').dialog('close');

                $('#btnLogout').click();
